643 void sp::ast_worker::process_formula(sp::ast *ast) {
644  DBG << "process sp::ast_worker::process_formula";
645  this->assert_ast_type(ast, 12, AST_CONDITION, AST_COND_QUERY, AST_COND_EQUAL, AST_COND_NOT_EQUAL, AST_COND_IN, AST_COND_NOT_IN, AST_COND_AND, AST_COND_FORALL, AST_COND_EXISTS, AST_COND_OR, AST_COND_NOT, AST_COND_BRACKETS);
646 
647  std::list<sp::ast*>::iterator it = ast->get_children()->begin();
648  switch (ast->get_type()) {
649  case AST_COND_AND:
650  case AST_COND_OR:
651  this->process_formula(*it++);
652  this->process_formula(*it++);
653  break;
654  case AST_COND_NOT:
655  case AST_COND_BRACKETS:
656  this->process_formula(*it++);
657  break;
658  case AST_COND_FORALL:
659  case AST_COND_EXISTS:
660  this->process_quantifier(*it++, *it++, *it++);
661  break;
662  case AST_COND_EQUAL:
663  case AST_COND_NOT_EQUAL:
664  this->assert_is_same_type(*it++, *it++);
665  break;
666  case AST_COND_IN:
667  case AST_COND_NOT_IN:
668  this->assert_is_elem_of(*it++, *it++);
669  break;
670  case AST_COND_QUERY:
671  /* check if queries are enabled. */
672  if (!dvr->are_queries_enabled()) {
673  ERR << "worker for '" << *this->name << "', line " << ast->get_line_num() << ": query encountered , but queries are disabled.";
674  this->successful = false;
675  return;
676  }
677  for (std::list<sp::ast*>::iterator id_it = (*++it)->get_children()->begin(); id_it != (*it)->get_children()->end(); ++id_it) {
678  this->assert_exists_id(this->process_val(*id_it), (*id_it)->get_line_num());
679  }
680  break;
681  case AST_CONDITION:
682  this->assert_valid_condition_call(*it++, *it++);
683  break;
684  default:
685  /* should not happen */
686  FTL << "worker: invalid point reached.";
687  exit(EXIT_FAILURE);
688  }

691 void sp::ast_worker::process_quantifier(sp::ast *id_ast, sp::ast *set_ast, sp::ast *formula_ast) {
692  DBG << "process sp::ast_worker::process_quantifier";
693  this->assert_ast_type(id_ast, 2, AST_VAL, AST_TUPLE);
694 
695  if (id_ast->get_type() == AST_VAL) {
696  std::string *id = this->process_val(id_ast);

699  if (set_ast->get_type() == AST_VAL) {
700  std::string *set_id = this->process_val(set_ast);
701  std::string *set = nullptr;
702  if (!this->assert_exists_id(set_id, set_ast->get_line_num())) {
703  return;
704  }
705 
706  /* checks if the set identifier is iterateable. */
707  if (this->temp_ids->find(set_id) == this->temp_ids->end()) {
708  set = set_id;
709  }
710  else if (this->temp_ids->at(set_id)->is_set()) {
711  set = this->temp_ids->at(set_id)->get_comp();
712  }
713  else {
714  ERR << "worker for '" << *this->name << "', line " << set_ast->get_line_num() << ": it is not possible to iterate over the identifier '" << *set_id << "'.";
715  this->successful = false;
716  }
717 
718  /* create a new identifier and process the contained condition */
719  this->temp_ids->insert(std::pair<std::string*, sp::parameter*>(id, new sp::parameter(set, false)));
720  this->process_formula(formula_ast);
721  this->temp_ids->erase(id);
722  }
723  else { /* set_ast is AST_FUNCTION */
724  std::list<sp::ast*>::iterator func_it = set_ast->get_children()->begin();

727  return;

733  if (map->is_mapping_set()) {
734  this->temp_ids->insert(std::pair<std::string*, sp::parameter*>(id, new sp::parameter( map->get_value_comp(), false)));
735  this->process_formula(formula_ast);
736  this->temp_ids->erase(id);
737  }
738  else {
739  ERR << "worker for '" << *this->name << "', line " << set_ast->get_line_num() << ": it is not possible to iterate over the value of the function '" << *func << "'.";
740  this->successful = false;
741  }
742  }
743  }
744  else { /* id_ast is AST_TUPLE */
745  this->assert_ast_type(set_ast, 1, AST_VAL);
746  std::string *set = this->process_val(set_ast);
747  this->assert_exists_id(set, set_ast->get_line_num());

749  this->assert_ast_type(id_list, 1, AST_VAL_LIST);
750  std::string *rel_name = nullptr;
751 
752  /* checks if the set identifier is iterateable. */
753  if (this->temp_ids->find(set) == this->temp_ids->end()) {
754  rel_name = set;
755  }
756  else if (this->temp_ids->at(set)->is_set()) {
757  rel_name = this->temp_ids->at(set)->get_comp();
758  }

760  ERR << "worker for '" << *this->name << "', line " << set_ast->get_line_num() << ": it is not possible to iterate over the identifier '" << *set << "'.";
761  this->successful = false;
762  }
