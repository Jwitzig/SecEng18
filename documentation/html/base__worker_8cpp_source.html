596  break;

598  this->process_assignment_mapping_intersection(*it++, *it++, *it++);

602  break;
603  case AST_STATEMENT_DEF_FUNCTION:

607  case AST_STATEMENT_FOR_TUPLE_DEF:

746 void sp::base_worker::process_for_assignment(sp::ast *id_ast, sp::ast *set_ast, sp::ast *statement_list_ast) {
747  DBG << "process sp::base_worker::process_for_assignment";
748  this->assert_ast_type(id_ast, 2, AST_VAL, AST_TUPLE);
749  this->assert_ast_type(statement_list_ast, 1, AST_STATEMENT_DEF_LIST);
750  std::string *set_id = this->process_val(set_ast);
751  this->assert_exists_id(set_id, set_ast->get_line_num());
752  std::string *comp = nullptr;
753 
754  /* checks if the set identifier is iterateable. */
755  if (this->temp_ids->find(set_id) == this->temp_ids->end()) {
756  comp = set_id;
757  }
758  else if (this->temp_ids->at(set_id)->is_set()) {
759  comp = this->temp_ids->at(set_id)->get_comp();
760  }
761  else {
762  ERR << "worker for '" << *this->name << "', line " << set_ast->get_line_num() << ": it is not possible to iterate over the identifier '" << *set_id << "'.";
763  this->successful = false;
764  return;
765  }
766 
767 
768  if (id_ast->get_type() == AST_VAL) {
769  std::string *id = this->process_val(id_ast);
770 
771  /* create a new identifier and process the inner statement list */
772  this->temp_ids->insert(std::pair<std::string*, parameter*>(id, new parameter(comp, false)));
773  for (std::list<sp::ast*>::iterator stmt_it = statement_list_ast->get_children()->begin(); stmt_it != statement_list_ast->get_children()->end(); ++stmt_it) {
774  this->process_statement(*stmt_it);
775  }
776  this->temp_ids->erase(id);
777  }
778  else { /* id_ast is AST_TUPLE */
779  sp::ast *id_list = *id_ast->get_children()->begin();
780 
781  if (!this->assert_exists_relation(comp, set_ast->get_line_num())) {
782  return;
783  }
784 
785  std::list<std::string*> *rel = this->base->get_relation(comp);
786  if (rel->size() != id_list->get_children()->size()) {
787  ERR << "worker for '" << *this->name << "', line " << id_list->get_line_num() << ": the tuple '" << *set_id << "' has an invalid element size, " << id_list->get_children()->size()
788  << " element(s) instead of " << rel->size() << " element(s).";
789  this->successful = false;
790  }
791 
792  /* create a new identifier for each element in the tuple. */
793  std::list<sp::ast*>::iterator id_it = id_list->get_children()->begin();
794  std::list<std::string*>::iterator rel_it = rel->begin();
795  while (id_it != id_list->get_children()->end() and rel_it != rel->end()) {
796  std::string *id = this->process_val(*id_it);
797  this->assert_unique_id(id, (*id_it++)->get_line_num());
798  this->temp_ids->insert(std::pair<std::string*, parameter*>(id, new parameter(*rel_it++, false)));
799  }
800  /* process the inner statement list. */
801  for (std::list<sp::ast*>::iterator stmt_it = statement_list_ast->get_children()->begin(); stmt_it != statement_list_ast->get_children()->end(); ++stmt_it) {
802  this->process_statement(*stmt_it);
803  }
804  for (std::list<sp::ast*>::iterator it = id_list->get_children()->begin(); it != id_list->get_children()->end(); ++it) {
805  this->temp_ids->erase(this->process_val(*it));
806  }
807  }
