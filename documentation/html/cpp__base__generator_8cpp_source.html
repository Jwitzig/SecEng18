40 void sp::cpp_base_generator::generate_components() {
41 
42  std::list<std::string*> setnames = base->get_sets_order();
43  for (auto& it : setnames) {
44 
45  components.push_back("typedef std::string " + *it + "_elem;");
46  components.push_back("typedef Set<" + *it + "_elem> " + *it + "_type;");
47  }
48 
49  std::list<std::string*> rel_order = base->get_relations_order();
50  for (auto& it : rel_order) {
51  std::list<std::string*> rel_comps = *(base->get_relation(it));
52  std::string temp = "typedef Relation<";
53  for (auto& it2 : rel_comps) {
54  temp.append(*it2 + "_elem,");
55  }
56  temp.pop_back();
57  temp.append("> " + *it + "_type;");
58  components.push_back(temp);
59  }
60  std::map<std::string*, mapping*, compare_string_pointer> mappings = *(base->get_all_mappings());
61  std::list<std::string*> map_order = base->get_mapping_order();
62  for (auto& it : map_order) {
63  sp::mapping* map_content = base->get_mapping(it);
64  std::string temp = "typedef ";
65  if (map_content->is_mapping_set()) {
66  temp.append("Map_set<");
67  } else {
68  temp.append("Map_elem<");
69  }
70  //mapped type of the mapping
71  temp.append(*(map_content->get_value_comp()) + "_elem");
72 
73  //add mapping params
74  std::list<std::string*> params = *(map_content->get_key_comps());
75  for (auto& it2 : params) {
76  temp.append("," + *it2 + "_elem");

97 std::string sp::cpp_base_generator::process_cond_body(sp::ast* ast) {
98  DBG << "process sp::cpp_base_generator::process_cond_body";
99  std::list<sp::ast*>::iterator it = ast->get_children()->begin();
100  switch (ast->get_type()) {
101  case AST_COND_AND:
102  case AST_COND_OR:
103  {
104  DBG << "sp::cppgenerator::process_cond_body - case: and, or";
105  if (ast->get_type() == AST_COND_AND) {
106  return (this->process_cond_body(*it++) + " && " + this->process_cond_body(*it++));
107  } else { /* ast is AST_COND_OR */
108  return (this->process_cond_body(*it++) + " || " + this->process_cond_body(*it++));
109  }
110  break;
111  }
112  case AST_COND_NOT:
113  {
114  DBG << "sp::cppgenerator::process_cond_body - case: not";
115  /* process the contained condition part */
116  return ("!(" + this->process_cond_body(*it++) + ")");
117 
118  break;
119  }
120  case AST_COND_BRACKETS:
121  {
122  DBG << "sp::cppgenerator::process_cond_body - case: ()";
123  /* process the contained condition part. */
124  return ( "(" + this->process_cond_body(*it++) + ")");
125  break;
126  }
127  case AST_COND_FORALL:
128  case AST_COND_EXISTS:
129  {
130  DBG << "sp::cppgenerator::process_cond_body - case: forall, exists";
131  std::string type;
132  std::string ret;
133 
134  if (ast->get_type() == AST_COND_FORALL) {
135  type = "std::all_of";
136  } else {
137  type = "std::any_of";
138  }
139 
140  sp::ast *id_ast = *it++; //elem
141  sp::ast *set_ast = *it++; //set...
142  sp::ast *inner_form = *it++; //cond
143 
144  std::string elem_name;
145  std::string set_name;
146  switch (id_ast->get_type()) {
147  case AST_VAL:
148  {
149 
150  std::string *id_val = ((ast_value*) id_ast)->get_value();
151  elem_name = *id_val;
152 
153  if (set_ast->get_type() == AST_VAL) {
154  set_name = *(((ast_value*) set_ast)->get_value());
155  } else { /* set_ast is AST_FUNCTION */
156  std::list<sp::ast*>::iterator func_it_elem = set_ast->get_children()->begin();
157  std::string func_name = extract_val(*func_it_elem++);
158  set_name = (func_name + "(std::make_tuple(" + gen_param_list(*func_it_elem) + "))");
159  }
160  ret = (type + "(" + set_name + ".begin()," + set_name + ".end(),"
161  + "[=] (auto " + elem_name + ") {return(" +
162  this->process_cond_body(inner_form)
163  + ");}" + ")");
164  break;
165 
166  }
167  case AST_TUPLE:
168  {
169  //The condition has to be split because the C++ compiler doesn't recognize parts of the tuple in the subcondition
170  //e.g. forall (r1,r2) in RH: r1==... and r2 == ... is impossible
171 
172 
173  sp::ast *tuple = *id_ast->get_children()->begin();
174  set_name = *(((ast_value*) set_ast)->get_value());
175  /*
176  Create a name for the tuple consisting of its element id's
177  */
178 
179  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
180  std::string id_val = extract_val(*tuple_it);
181  elem_name.append(id_val);
182 
183  }
184  int count = 0;
185 
186  //Create link between tuple and tuple elements
187  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
188  std::string *id_val = ((ast_value*) * tuple_it)->get_value();
189 
190  tuple_links.emplace(*id_val, "std::get<" + std::to_string(count) + ">(" + elem_name + ")");
191  count++;
192  }
193  ret = (type + "(" + set_name + ".begin()," + set_name + ".end(),"
194  + "[=] (auto " + elem_name + ") {return(" +
195  this->process_cond_body(inner_form)
196  + ");}" + ")");
197 
198  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
199 
200  std::string *id_val = ((ast_value*) * tuple_it)->get_value();
201  tuple_links.erase(*id_val);
202 
203  }
204  break;
205  }
206  default:
207  {
208  FTL << "error at condition forall/exists. shouldnt have happened";
209  EXIT_FAILURE;
210  }
211  }
212 
213 
214  return ret;
215  break;
216  }
217  case AST_COND_EQUAL:
218  case AST_COND_NOT_EQUAL:
219  {
220  DBG << "sp::cppgenerator::process_cond_body - case: eq, ne";
221  /* assigns the result to a newly created identifier. */
222  /* definition should replaced in the process. */
223  std::string equal_op;
224  if (ast->get_type() == AST_COND_EQUAL) {
225  equal_op = "==";
226  } else {
227  equal_op = "!=";
228  }
229  sp::ast *elem_1 = *it++;
230 
231  sp::ast *elem_2 = *it++;
232 
233  std::string expr1;
234  std::string expr2;
235 
236  std::array<sp::ast*, 2> temp_pair = {elem_1, elem_2};
237  std::vector<std::string> expr_list = {expr1, expr2};
238  /* tuples are compared. */
239 
240  for (int i = 0; i < 2; i++) {
241 
242  switch (temp_pair[i]->get_type()) {
243  case AST_TUPLE:
244  case AST_TUPLE_VALUE:
245  {
246 
247  expr_list[i] = "std::make_tuple(" + gen_param_list(temp_pair[i]) + ")";
248  break;
249  }
250  case AST_VAL:
251  case AST_VALUE:
252  {
253 
254  expr_list[i] = this->extract_val(temp_pair[i]);
255  break;
256  }
257  case AST_FUNCTION:
258  {
259  std::list<sp::ast*>::iterator func_it_elem = temp_pair[i]->get_children()->begin();
260  /* process the function parameters. */
261  std::string func_name = extract_val(*func_it_elem++);
262  expr_list[i] = (func_name + "(std::make_tuple(" + this->gen_param_list(*func_it_elem) + "))");
263  break;
264  }
265  default:
266  {
267  FTL << "Invalid Comparator in Formula - EQ/NEQ";
268  exit(EXIT_FAILURE);
269  }
270  }
271 
272  }
273  return (expr_list[0] + equal_op + expr_list[1]);
274  break;
275  }
276 
277 
278  case AST_COND_IN:
279  case AST_COND_NOT_IN:
280  {
281  DBG << "sp::cppenerator::process_cond_body - case: in, not in";
282  /* assigns the result to a newly created identifier. */
283  sp::ast *elem = *it++;
284  sp::ast *containing_elem = *it++;
285  std::string element;
286  std::string container;
287  std::array <sp::ast*, 2> typetest = {elem, containing_elem};
288  std::array <std::string, 2> returnpair;
289  /* result of functions is comprised in a function. */
290  for (int i = 0; i < 2; i++) {
291  switch (typetest[i]->get_type()) {
292  case AST_FUNCTION:
293  {
294  std::list<sp::ast*>::iterator func_it_elem = typetest[i]->get_children()->begin();
295  /* process the function parameters. */
296  std::string func_name = extract_val(*func_it_elem++);
297 
298  returnpair[i] = (func_name + "(std::make_tuple(" + gen_param_list(*func_it_elem) + "))");
299  break;
300  }
301  case AST_VAL:
302  case AST_VALUE:
303  {
304  returnpair[i] = this->extract_val(typetest[i]);
305  break;
306  }/* a tuple is comprised. */
307  case AST_TUPLE:
308  case AST_TUPLE_VALUE:
309  { /* if ((elem->get_type() == AST_TUPLE or elem->get_type() == AST_TUPLE_VALUE) and containing_elem->get_type() == AST_VAL) */
310  returnpair[i] = "std::make_tuple(" + gen_param_list(typetest[i]) + ")";
311  break;
312  }
313  default:
314  {
315  FTL << "invalid Argument in Formulat: in";
316  exit(EXIT_FAILURE);
317  }
318  }
319  }
320  if (ast->get_type() == AST_COND_IN) {
321  return (returnpair[1] + ".in(" + returnpair[0] + ")");
322  } else {
323  return ("!(" + returnpair[1] + ".in(" + returnpair[0] + "))");
324  }
325  break;
326  }
327  case AST_COND_QUERY:
328  {
329  DBG << "sp::cppgenerator::process_cond_body - case: query";
330  /* assigns the result to a newly created identifier. */
331  std::string query = this->extract_val(*(*it++)->get_children()->begin());
332  /* process the parameters for the query. */
333  query.append("(");
334  for (std::list<sp::ast*>::iterator id_it = (*it)->get_children()->begin(); id_it != (*it)->get_children()->end(); ++id_it) {
335  query.append(this->extract_val(*id_it) + ",");
336  }
337  query.pop_back();
338  query.append(")");
339  return query;
340  break;
341  }
342  case AST_CONDITION:
343  {
344  DBG << "sp::cppgenerator::process_cond_body - case: condition";
345  /* call a condition and insert the result to the formula. */
346  //condition originates from base policy
347  return extract_val(*it++) + "(" + gen_param_list(*it++) + ")";
348  break;
349  }
350  default:
351  {
352  FTL << "Error at formula. No valid Subtree";
353  /* should not happen */
354  FTL << "cppgenerator: invalid point reached.";
355  exit(EXIT_FAILURE);
356  break;
357  }
358  }

375 std::string sp::cpp_base_generator::process_prim_body(sp::ast *ast) {
376  DBG << "process sp::cppgenerator::generate_prim_ast";
377  std::string ret;
378  for (std::list<sp::ast*>::iterator it = ast->get_children()->begin(); it != ast->get_children()->end(); ++it) {
379  std::list<sp::ast*>::iterator part_it = (*it)->get_children()->begin();
380  switch ((*it)->get_type()) {
381  case AST_STATEMENT_DEF_UNION_SAME:
382  {
383  DBG << "sp::cppgenerator::generate_prim_ast - case: union same";
384  std::string *comp = ((ast_value*) * part_it++)->get_value();
385  /* Use iterators to union the containers */
386  std::string set_1 = this->extract_val(*part_it++);
387  std::string set_2 = this->extract_val(*part_it++);
388  ret.append("std::set_union("
389  + set_1 + ".begin()," + set_1 + ".end(),"
390  + set_2 + ".begin()," + set_2 + ".end(),"
391  + set_1 + ".begin());");
392 
393  break;
394  }
395  case AST_STATEMENT_DEF_INTERSECTION_SAME:
396  {
397  DBG << "sp::cppgenerator::generate_prim_ast - case: intersection same";
398  std::string *comp = ((ast_value*) * part_it++)->get_value();
399  /* Use the iterators to intersect */
400 
401  std::string set_1 = this->extract_val(*part_it++);
402  std::string set_2 = this->extract_val(*part_it++);
403  ret.append("std::set_difference("
404  + set_1 + ".begin(), " + set_1 + ".end(), "
405  + set_2 + ".begin(), " + set_2 + ".end(), "
406  + set_1 + ".begin());");
407  break;
408  }
409  case AST_STATEMENT_DEF_UNION_SET:
410  {
411  DBG << "sp::cppgenerator::generate_prim_ast - case: union set";
412  std::string set = this->extract_val(*part_it++);
413  part_it++;
414  std::string elem = this->extract_val(*part_it++);
415  ret.append(set + ".insert(" + elem + ");");
416  break;
417  }
418  case AST_STATEMENT_DEF_INTERSECTION_SET:
419  {
420  DBG << "sp::cppgenerator::generate_prim_ast - case: intersection set";
421  std::string set = this->extract_val(*part_it++);
422  part_it++;
423  std::string elem = this->extract_val(*part_it++);
424  ret.append(set + ".erase(" + elem + ");");
425  break;
426  }
427  case AST_STATEMENT_DEF_UNION_RELATION:
428  {
429  DBG << "sp::cppgenerator::generate_prim_ast - case: add tuple to relation";
430  std::string relation = this->extract_val(*part_it++);
431  part_it++;
432  std::string tuple = this->gen_param_list(*part_it);
433  ret.append(relation + ".insert(std::make_tuple(" + tuple + "));");
434  break;
435  }
436  case AST_STATEMENT_DEF_INTERSECTION_RELATION:
437  {
438  DBG << "sp::cppgenerator::generate_prim_ast - case: remove tuple from relation";
439  std::string relation = this->extract_val(*part_it++);
440  part_it++;
441  std::string tuple = this->gen_param_list(*part_it);
442  ret.append(relation + ".erase(std::make_tuple(" + tuple + "));");
443  break;
444  }
445  case AST_STATEMENT_DEF_UNION_MAPPING:
446  {
447  DBG << "sp::cppgenerator::generate_prim_ast - case: union mapping";
448  //"insert-into-mapping"
449  std::string map = this->extract_val(*part_it++);
450  part_it++;
451  /* adds the key for the mapping elem. */
452  std::string map_keys = gen_param_list(*part_it++);
453  /* adds the value. */
454  std::string value = this->extract_val(*part_it);
455  ret.append(map + ".insert(std::make_tuple(" + map_keys + ")," + value + ");");
456  break;
457  }
458  case AST_STATEMENT_DEF_INTERSECTION_MAPPING:
459  {
460  DBG << "sp::cppgenerator::generate_prim_ast - case: remove-from-mapping";
461  std::string map = this->extract_val(*part_it++);
462  part_it++; //this ASR has the mapped value to be erased. since there can only one key with same name its not required to delete the pair.
463  ret.append(map + ".erase(std::make_tuple(" + gen_param_list(*part_it) + "));");
464  break;
465  }
466  case AST_STATEMENT_DEF_UNION_MAPPING_SET:
467  {
468  DBG << "sp::cppgenerator::generate_prim_ast - case: insert-into-mapping";
469  //"insert-into-mapping" e.g. insert set of values into given key (key is tuple);
470  std::string map = this->extract_val(*part_it++);
471  /* adds the key for the mapping elem. */
472  std::string map_keys = "std::make_tuple(" + gen_param_list(*part_it++) + ")";
473  /* create set and add as many values as given. */
474  std::string values = gen_param_list(*part_it++);
475 
476  ret.append(map + ".insert(" + map_keys + "," + values + ");");
477  break;
478  }
479  case AST_STATEMENT_DEF_INTERSECTION_MAPPING_SET:
480  {
481  DBG << "sp::cppgenerator::generate_prim_ast - case: remove-from-mapping-set";
482  std::string map = this->extract_val(*part_it++);
483  part_it++;
484  /* adds the key for the mapping elem. */
485  std::string map_keys = gen_param_list(*part_it++);
486  /* adds the value. */
487  std::string value = this->extract_val(*part_it);
488  ret.append(map + ".erase(std::make_tuple(" + map_keys + ")," + value + ");");
489  break;
490  }
491  case AST_STATEMENT_DEF_FUNCTION:
492  {
493  DBG << "sp::cppgenerator::generate_prim_ast - case: function";
494  std::string id = *((ast_value*) * part_it++)->get_value();
495  std::string value = *((ast_value*) * part_it++)->get_value();
496  std::string params = gen_param_list(*part_it);
497  /* generate new identifier definition with the value of the mapping (with the given parameters). */
498  ret.append("auto " + id + " = " + value + "(std::make_tuple(" + params + "));");
499  break;
500  }
501  case AST_STATEMENT_FOR_DEF:
502  {
503  DBG << "sp::cppgenerator::generate_prim_ast - case: for";
504  std::string *id_val = ((ast_value*) * part_it++)->get_value();
505  std::string set = this->extract_val(*part_it++);
506  sp::ast *inner_stmts = *part_it++;
507  ret.append("for( auto& " + *id_val + ": " + set + "){" + process_prim_body(inner_stmts) + "}");
508  break;
509  }
510 
511  case AST_STATEMENT_FOR_TUPLE_DEF:
512  {
513  DBG << "sp::cppgenerator::generate_prim_ast - case: for tuple";
514  sp::ast *tuple = *(*part_it++)->get_children()->begin();
515  std::string rel = this->extract_val(*part_it++);
516  sp::ast *inner_stmts = *part_it++;
517 
518 
519  //The condition has to be split because the C++ compiler doesn't recognize parts of the tuple in the subcondition
520  //e.g. forall (r1,r2) in RH: r1==... and r2 == ... is impossible
521 
522  std::string iterator_name;
523 
524  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
525  std::string id_val = extract_val(*tuple_it);
526  iterator_name.append(id_val);
527 
528  }
529  int count = 0;
530 
531  //Create link between tuple and tuple elements
532  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
533  std::string *id_val = ((ast_value*) * tuple_it)->get_value();
534 
535  tuple_links.emplace(*id_val, "std::get<" + std::to_string(count) + ">(" + iterator_name + ")");
536  count++;
537  }
538  ret = ("for (auto& " + iterator_name + " : " + rel + "){ \r"
539  + this->process_cond_body(inner_stmts)
540  + "\r }");
541 
542  for (std::list<sp::ast*>::iterator tuple_it = tuple->get_children()->begin(); tuple_it != tuple->get_children()->end(); ++tuple_it) {
543 
544  std::string *id_val = ((ast_value*) * tuple_it)->get_value();
545  tuple_links.erase(*id_val);
546 
547  }
548  break;
549  }
550 
551  case AST_PRIMITIVE:
552  {
553  DBG << "sp::cppgenerator::generate_prim_ast - case: primitive";
554  /* call a primitive and insert the result to the statement list. */
555  ret.append(extract_val(*part_it++) + "(" + gen_param_list(*part_it++) + ");");
556 
557  break;
558  }
559  default:
560  {
561  /* should not happen */
562  FTL << "cppgenerator: invalid point reached.";
563  exit(EXIT_FAILURE);
564  }
565  }
566  ret.append(" \r ");
567  }
568  return ret;
569 }

571 std::string sp::cpp_base_generator::extract_val(ast *ast) {
572  DBG << "process sp::cpp_base_generator::generate_node_val";
573  switch (ast->get_type()) {
574  case AST_VALUE:
575  {
576  std::string value = *(((sp::ast_value*) * ast->get_children()->begin())->get_value());
577  if (tuple_links.count(value) == 1) {
578  value = "\"" + tuple_links.at(value) + "\"";
579  } else {
580  value = "\"" + *(((sp::ast_value*) * ast->get_children()->begin())->get_value()) + "\"";
581  }
582  return value;
583  break;
584  }
585  case AST_VAL:
586  {
587  std::string value = *(((sp::ast_value*) ast)->get_value());
588  if (tuple_links.count(value) == 1) {
589  value = tuple_links.at(value);
590  }
591  return (value);
592  break;
593  }
594  case AST_FUNCTION:
595  { //just in case
