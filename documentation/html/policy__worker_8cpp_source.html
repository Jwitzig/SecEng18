117  for (std::list<sp::ast*>::iterator it = ast->get_children()->begin(); it != ast->get_children()->end(); ++it) {
118  this->assert_ast_type(*it, 3, AST_COMMAND_WITHOUT_CONDITION, AST_COMMAND_WITHOUT_BODY, AST_COMMAND);

120  std::string *name = this->process_val(*command_it);
121 
122  this->assert_unique_id(name, (*command_it)->get_line_num());
123  if (this->commands->find(name) != this->commands->end()) {
124  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the command '" << *name << "' already exists.";
125  this->successful = false;
126  }
127  this->commands->insert(name);
128  this->assert_ast_type(*++command_it, 1, AST_PARAMETER_LIST);
129 
130  /* save the id and type of every parameter of the command. */
131  for (std::list<sp::ast*>::iterator param_it = (*command_it)->get_children()->begin(); param_it != (*command_it)->get_children()->end(); ++param_it) {
132  this->assert_ast_type(*param_it, 2, AST_ELEM_PARAMETER, AST_SET_PARAMETER);
133

136  this->assert_exists_component(comp, (*it++)->get_line_num());

138  this->assert_unique_id(id, (*it)->get_line_num());
139

142  param = new parameter(comp, true);
143  }

145  param = new parameter(comp, false);
146  }
147  else {
148  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the component '" << *comp << "' is not part of the input vector.";

154 
155  }
156

159  this->process_formula(*++command_it);
160  }
161

187 void sp::policy_worker::process_initial_values(sp::ast* ast) {
188  DBG << "process sp::policy_worker::process_initial_state";
189  this->assert_ast_type(ast, 2, AST_INITIAL_STATE_LIST, AST_EXTENSION_VECTOR_LIST);
190 
191  /* for every model component */
192  for (std::list<sp::ast*>::iterator state_it = ast->get_children()->begin(); state_it != ast->get_children()->end(); ++state_it) {
193  this->assert_ast_type(*state_it, 5, AST_STATE_SET, AST_STATE_RELATION, AST_STATE_MAPPING_ELEM, AST_STATE_MAPPING_SET, AST_EMPTY_LIST);
194  std::list<sp::ast*>::iterator it = (*state_it)->get_children()->begin();
195  std::string *comp = this->process_val(*it);
196  /* if a component is already initiated. */
197  if (this->temp_ids->find(comp) != this->temp_ids->end()) {
198  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the component '" << *comp << "' was already initialised.";
199  this->successful = false;
200  }
201  this->temp_ids->insert(std::pair<std::string*, parameter*>(comp, nullptr));
202  /* if the initial state is processed and an element of the extension vector is encounter. */
203  if (ast->get_type() == AST_INITIAL_STATE_LIST and this->state_components->find(comp) == this->state_components->end()) {
204  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the component " << *comp << "' is not contained in the state set.";
205  this->successful = false;
206  }
207  /* if the extension vector is processed and an element of the state space is encounter. */
208  else if (ast->get_type() == AST_EXTENSION_VECTOR_LIST and this->state_components->find(comp) != this->state_components->end()) {
209  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the component " << *comp << "' is contained in the state set.";
210  this->successful = false;
211  }
212  else if ((*state_it)->get_type() == AST_STATE_SET) {
213  this->assert_exists_set(comp, (*it)->get_line_num());
214  }
215  else if ((*state_it)->get_type() == AST_STATE_RELATION) {
216  this->assert_ast_type((*++it), 1, AST_TUPLE_VALUE_LIST);
217  if (!this->assert_exists_relation(comp, (*it)->get_line_num())) {
218  return;
219  }
220  int rel_size = this->base->get_relation(comp)->size();
221  /* for every tuple. */
222  for (std::list<sp::ast*>::iterator tuple_it = (*it)->get_children()->begin(); tuple_it != (*it)->get_children()->end(); ++tuple_it) {
223  this->assert_ast_type(*tuple_it, 1, AST_TUPLE_VALUE);
224  /* check if the tuple size is correct. */
225  int tuple_size = (*(*tuple_it)->get_children()->begin())->get_children()->size();
226  if (rel_size != tuple_size) {
227  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the tuple of the relation '" << *comp << "' has an invalid element size, "
228  << tuple_size << " element(s) instead of " << rel_size << " element(s).";
229  this->successful = false;
230  }
231  }
232  }
233  else if ((*state_it)->get_type() == AST_STATE_MAPPING_ELEM) {
234  this->assert_ast_type((*++it), 1, AST_MAPPING_VALUE_LIST);
235  if (!this->assert_exists_mapping(comp, (*it)->get_line_num())) {
236  return;
237  }
238  int map_comp_size = this->base->get_mapping(comp)->get_key_comps()->size();
239  /* for every mapping element. */
240  for (std::list<sp::ast*>::iterator map_it = (*it)->get_children()->begin(); map_it != (*it)->get_children()->end(); ++map_it) {
241  this->assert_ast_type(*map_it, 1, AST_MAPPING_VALUE);
242  /* check if the key size of the mapping is correct. */
243  int value_key_size = (*(*map_it)->get_children()->begin())->get_children()->size();
244  if (map_comp_size != value_key_size) {
245  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the key of the mapping '" << *comp << "' has an invalid element size, "
246  << value_key_size << " element(s) instead of " << map_comp_size << " element(s).";
247  this->successful = false;
248  }
249  }
250  }
251  else if ((*state_it)->get_type() == AST_STATE_MAPPING_SET) {
252  this->assert_ast_type((*++it), 1, AST_MAPPING_VALUE_SET_LIST);
253  if (!this->assert_exists_mapping_set(comp, (*it)->get_line_num())) {
254  return;
255  }
256  int map_comp_size = this->base->get_mapping(comp)->get_key_comps()->size();
257  /* for every mapping element. */
258  for (std::list<sp::ast*>::iterator map_it = (*it)->get_children()->begin(); map_it != (*it)->get_children()->end(); ++map_it) {
259  this->assert_ast_type(*map_it, 1, AST_MAPPING_VALUE_SET);
260  /* check if the key size of the mapping is correct. */
261  int value_key_size = (*(*map_it)->get_children()->begin())->get_children()->size();
262  if (map_comp_size != value_key_size) {
263  ERR << "worker for '" << *this->name << "', line " << (*it)->get_line_num() << ": the key of the mapping '" << *comp << "' has an invalid element size, "
264  << value_key_size << " element(s) instead of " << map_comp_size << " element(s).";
265  this->successful = false;
266  }
267  }
268  }
269  /*
270  AST_EMPTY_LIST still possible.
271  in this case no checks are necessary.
272  */
273  }

276 void sp::policy_worker::check_state_modifications(sp::ast* ast) {
277  /* for every statement */
278  for (std::list<sp::ast*>::iterator it = ast->get_children()->begin(); it != ast->get_children()->end(); ++it) {
279  sp::ast_type type = (*it)->get_type();
280  /* ignore the statement if it is a function */
281  if (type != AST_STATEMENT_DEF_FUNCTION) {
282  if (type == AST_STATEMENT_FOR_DEF or type == AST_STATEMENT_FOR_TUPLE_DEF) {
283  std::list<sp::ast*>::iterator for_it = (*it)->get_children()->begin();
284  ++for_it; /* element */
285  ++for_it; /* set */
286  this->check_state_modifications(*for_it); /* statement list */
287  }
288  else if (type == AST_PRIMITIVE) {
289  this->check_state_modifications(*(*it)->get_children()->begin());
290  }
291  else {
292  std::string *comp = ((ast_value*)*(*it)->get_children()->begin())->get_value();
293  if (this->base->name_exists(comp)) {
294  if (this->state_components->find(comp) == this->state_components->end()) {
295  ERR << "worker for '" << *this->name << "', line " << ast->get_line_num() << ": the component '" << *comp << "' is part of the extension vector but it is modified by a primitive.";
296  this->successful = false;
297  }
298  else {
299  this->modifiable_state_components->insert(comp);
300  }
301  }
302  }
303  }
304  }
